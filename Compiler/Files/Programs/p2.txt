let n : int ;
input ( n ) ;

let p : string ;
p = " p " ;

let e : string ;
e = " e " ;

if ( n < 2 ) {
    output ( e ) ;
    return ;
}

if ( n == 2 ) {
    output ( p ) ;
    return ;
}

if ( n % 2 == 0 ) {
    output ( e ) ;
    return ;
}

let d : int ;
d = 3 ;

while ( d * d <= n ) {
    if ( n % d == 0 ) {
        output ( e ) ;
        return ;
    }

    d = d + 2 ;
}

output ( p ) ;
